<?php

// Inicia o reanuda la sesion existente
session_start();

// Comprueba si la variable de sesión 'usuario_id' NO existe.
if (!isset($_SESSION['usuario_id'])) {
    // Si no existe, significa que el usuario no ha iniciado sesión.
    // Lo redirigimos a la página de login.
    header('Location: login_usuario.html');
    exit(); // Es importante detener la ejecución del resto de la página.
}
?>
